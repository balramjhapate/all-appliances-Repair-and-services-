2.2 GCLID Capture Code (landing page mein daalo)
// lib/tracking.ts
export function captureGCLID() {
  const params = new URLSearchParams(window.location.search);
  const gclid = params.get('gclid');
  if (gclid) {
    localStorage.setItem('gclid', gclid);
    localStorage.setItem('gclid_time', new Date().toISOString());
    localStorage.setItem('utm_source', params.get('utm_source') || '');
    localStorage.setItem('utm_campaign', params.get('utm_campaign') || '');
    localStorage.setItem('utm_keyword', params.get('utm_keyword') || '');
    localStorage.setItem('landing_page', window.location.pathname);
  }
}

export function getStoredTracking() {
  return {
    gclid: localStorage.getItem('gclid'),
    utm_source: localStorage.getItem('utm_source'),
    utm_campaign: localStorage.getItem('utm_campaign'),
    utm_keyword: localStorage.getItem('utm_keyword'),
    landing_page: localStorage.getItem('landing_page'),
  };
}
2.3 Lead Form with Tracking
// components/LeadForm.tsx
import { captureGCLID, getStoredTracking } from '@/lib/tracking';
import { supabase } from '@/lib/supabase';

export default function LeadForm({ city, service }) {
  useEffect(() => { captureGCLID(); }, []);

  const handleSubmit = async (e) => {
    const tracking = getStoredTracking();
    await supabase.from('leads').insert({
      name: e.target.name.value,
      phone: e.target.phone.value,
      city: city,
      service: service,
      device: /Mobile/.test(navigator.userAgent) ? 'mobile' : 'desktop',
      ...tracking
    });
    // Also fire Google Ads conversion
    window.gtag('event', 'conversion', {
      send_to: 'AW-XXXXXXXXX/YYYYYY',
      value: 1.0
    });
  };
}
